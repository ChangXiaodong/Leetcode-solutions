## 前序遍历

### 非递归
```python
def preorder_traversal(root):
    if not root:
        return
    nodeStack = []
    node = root
    res = []
    while node or nodeStack:
        while node:
            res.append(node.val)
            nodeStack.append(node)
            node = node.left
        node = nodeStack.pop()
        node = node.right
    return res
```
### 递归
```Python
def preorder_traversal_recursive(root, res):
    if not root:
        return
    res.append(root.val)
    if root.left:
        preorder_traversal_recursive(root.left, res)
    if root.right:
        preorder_traversal_recursive(root.right, res)
```

## 后序遍历

### 非递归
```python
def postorder_traversal(root):
    if not root:
        return
    nodeStack1 = [root]
    nodeStack2 = []
    res = []
    while nodeStack1:
        node = nodeStack1.pop()
        if node.left:
            nodeStack1.append(node.left)
        if node.right:
            nodeStack1.append(node.right)
        nodeStack2.append(node)
    while nodeStack2:
        node = nodeStack2.pop()
        res.append(node.val)
    return res
```

### 递归
```python
def postorder_traversal_recursive(root, res):
    if not root:
        return
    if root.left:
        postorder_traversal_recursive(root.left, res)
    if root.right:
        postorder_traversal_recursive(root.right, res)
    res.append(root.val)
```

## 按层遍历

### 非递归
```python
def levelorder_traversal(root):
    if not root:
        return
    nodeStack = [root]
    res = []
    while nodeStack:
        node = nodeStack.pop(0)
        res.append(node.val)
        if node.left:
            nodeStack.append(node.left)
        if node.right:
            nodeStack.append(node.right)
```
###  递归
```python
def dfs(node, res, level):
    if node:
        if len(res) < level + 1:
            res.append([])
        res[level].append(node.val)
        dfs(node.left, res, level + 1)
        dfs(node.right, res, level + 1)
```

## 路径保存

### 非递归
```python
def listAllPath(root):
    if root:
        return
    stack = [(root, [root.val])]
    res = []
    while stack:
        node, path = stack.pop()
        if not node.left and not node.right:
            res.append(path)
        if node.left:
            stack.append(node.left, path + [node.left.val])
        if node.right:
            stack.append(node.right, path + [node.right.val])
    return res
```

### 递归
```python
def dfs(node, path, res):
    if not node.left and not node.right:
        res.append(path)
    if node.left:
        dfs(node.left, path + [node.left.val], res)
    if node.right:
        dfs(node.right, path + [node.right.val], res)

def listAllPath(root):
    if not root:
        return
    res = []
    dfs(root, [root.val], res)
    return res
```


## 判断是否是对称二叉树
### 递归
```python
def isSymmetic_recursive(left, right):
    if not left and not right:
        return True
    if not left or not right:
        return False
    return left.val == right.val and \
           isSymmetic_recursive(left.left, right.right) and \
           isSymmetic_recursive(left.right, right.left)

def symmetric_tree(node):
    if not node:
        return
    return isSymmetic_recursive(node.left, node.right)
```

### 非递归
```python
def symmetric_tree(node):
    if not node:
        return
    stack = [node.left, node.right]
    while stack:
        right = stack.pop()
        left = stack.pop()
        if not left and not right:
            return True
        if not left or not right:
            return False
        if right.val == left.val:
            return True
        stack.append(left.left)
        stack.append(right.right)
        stack.append(left.right)
        stack.append(right.left)
    return True
```
